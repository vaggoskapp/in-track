async function updateData() { const timestamp = Date.now(); const targetUrl = `https://live.racefacer.com/ajax/live-data?slug=${trackName}&nocache=${timestamp}`; try { const response = await fetch(PROXY_URL + encodeURIComponent(targetUrl)); const data = await response.json(); const runners = data.data.runs; const myIndex = runners.findIndex(p => p.name.toLowerCase().includes(teamName.toLowerCase())); if (myIndex !== -1) { const myData = runners[myIndex]; document.getElementById('lastTime').innerText = myData.last_time || "0.000"; document.getElementById('posDisp').innerText = `P${myIndex + 1}`; document.getElementById('driverNameDisp').innerText = myData.name; // --- GAP AHEAD (Διαφορά από τον μπροστινό) --- const gapAheadEl = document.getElementById('gapAhead'); if (myIndex > 0) { const aheadData = runners[myIndex - 1]; let diffAhead; // 1. Προσπάθησε να πάρεις το έτοιμο "int" (interval) if (myData.int && myData.int !== "-") { diffAhead = parseFloat(myData.int.replace('+', '')); } // 2. Αν δεν υπάρχει, υπολόγισε από το last_passing else { diffAhead = Math.abs(myData.last_passing - aheadData.last_passing) / 1000; } // Έλεγχος για Laps if (diffAhead > 120 || (parseInt(aheadData.total_laps) > parseInt(myData.total_laps))) { let lapDiff = parseInt(aheadData.total_laps) - parseInt(myData.total_laps); gapAheadEl.innerText = `+${lapDiff} LAP${lapDiff > 1 ? 'S' : ''}`; } else { gapAheadEl.innerText = "+" + diffAhead.toFixed(2); if (lastGapAheadVal !== null) { gapAheadEl.style.color = diffAhead < lastGapAheadVal ? "#28a745" : "#ffc107"; } lastGapAheadVal = diffAhead; } } else { gapAheadEl.innerText = "LEADER"; gapAheadEl.style.color = "#fff"; } // --- GAP BEHIND (Διαφορά από τον πίσω) --- const gapBehindEl = document.getElementById('gapBehind'); if (myIndex < runners.length - 1) { const behindData = runners[myIndex + 1]; let diffBehind; // 1. Προσπάθησε να πάρεις το "int" του ΠΙΣΩ οδηγού if (behindData.int && behindData.int !== "-") { diffBehind = parseFloat(behindData.int.replace('+', '')); } // 2. Fallback στο last_passing else { diffBehind = Math.abs(behindData.last_passing - myData.last_passing) / 1000; } if (diffBehind > 120 || (parseInt(myData.total_laps) > parseInt(behindData.total_laps))) { let lapDiff = parseInt(myData.total_laps) - parseInt(behindData.total_laps); gapBehindEl.innerText = `+${lapDiff} LAP${lapDiff > 1 ? 'S' : ''}`; } else { gapBehindEl.innerText = "+" + diffBehind.toFixed(2); if (lastGapBehindVal !== null) { gapBehindEl.style.color = diffBehind > lastGapBehindVal ? "#28a745" : "#ffc107"; } lastGapBehindVal = diffBehind; } } else { gapBehindEl.innerText = "LAST"; gapBehindEl.style.color = "#fff"; } const isPit = myData.in_pit || String(myData.run_status).toLowerCase() === 'pit'; document.getElementById('status').innerText = isPit ? "IN PIT" : "LIVE LAP"; document.getElementById('lastTime').style.color = isPit ? "#333" : "#28a745"; } } catch (e) { console.log("API Error"); } }
